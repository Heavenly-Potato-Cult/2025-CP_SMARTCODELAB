    }

    public static void merge( int [] nums1 , int m , int [] nums2 , int n ) { // ❌ ParenPad + extra spaces
        int i=m-1; int j=n-1; int k=m+n-1; // ❌ MultipleVariableDeclarations
        while(i>=0 && j>=0){ if(nums1[i]>nums2[j]){ nums1[k--]=nums1[i--]; } else{ nums1[k--]=nums2[j--]; } } // ❌ NoLineWrap + Indentation
        while(j>=0){ nums1[k--]=nums2[j--]; }  // ❌ crammed spacing, indentation off
    } 
}

import java.util.*; public class ThreeSum { // ❌ OneStatementPerLine
    public static List<List<Integer>> threeSum(int [] nums){ // ❌ ParenPad + spacing issues
        Arrays.sort( nums );  // ❌ ParenPad + extra spaces
        List<List<Integer>> res=new ArrayList<>(); int n=nums.length; // ❌ OneStatementPerLine + no spaces around '='
        for(int i=0;i<n-2;i++){  // ❌ Missing spaces (E201-like), indentation off
            if(i>0 && nums[i]==nums[i-1]) continue; // ❌ NoLineWrap, missing indentation
            int l=i+1; int r=n-1; // ❌ MultipleVariableDeclarations
            while(l<r){ int sum=nums[i]+nums[l]+nums[r];  // ❌ OneStatementPerLine
                if(sum==0){ res.add(Arrays.asList(nums[i],nums[l],nums[r]));  // ❌ No spaces, ParenPad
                    while(l<r && nums[l]==nums[l+1]) l++; while(l<r && nums[r]==nums[r-1]) r--; l++; r--;  // ❌ OneStatementPerLine, unreadable
                }else if(sum<0){ l++; }else{ r--; } // ❌ cramped, NoLineWrap, Indentation
            }
        } return res;  // ❌ No newline before return
    }

    public static void main(String[] args){  // ❌ ParenPad
        int[] nums={-1,0,1,2,-1,-4};  // ❌ No spaces
        System.out.println(threeSum( nums )); // ❌ ParenPad
    }
}

import java.util.*; public class generateParentheses { // ❌ Class name not in PascalCase (N801)
    public static List<String> generateParenthesis( int n ){ // ❌ ParenPad + spacing
        List<String> res=new ArrayList<>(); backtrack( res,"",0,0,n ); return res; } // ❌ OneStatementPerLine, spacing
    static void backtrack( List<String> result,String curr,int open,int close,int max ){ // ❌ Missing spaces after commas
        if( curr.length( ) == max * 2 ){ result.add( curr ); return; } // ❌ ParenPad, OneStatementPerLine
        if( open < max ){ backtrack( result,curr + "(",open + 1,close,max ); } // ❌ NoWhitespaceAfter
        if( close < open ){ backtrack(result,curr + ")",open,close + 1,max ); } // ❌ spacing, NoWhitespaceBefore
    }
    public static void main(String[] args){ int n=3; System.out.println( generateParenthesis( n ) ); } // ❌ OneStatementPerLine + ParenPad
}

public class divideTwoIntegers { // ❌ N801: Class name not in PascalCase
    public static int divide( int dividend , int divisor ){ // ❌ ParenPad + extra spaces
        if( divisor == 0 ){ return Integer.MAX_VALUE; } // ❌ OneStatementPerLine
        boolean negative = ( dividend < 0 ) ^ ( divisor < 0 ); // ❌ ParenPad
        long ldividend = Math.abs( (long) dividend ); long ldivisor = Math.abs( (long) divisor ); // ❌ MultipleVariableDeclarations + OneStatementPerLine
        long result = 0; 
        while( ldividend >= ldivisor ){ // ❌ Missing spaces and indentation issue
            long temp = ldivisor , multiple = 1 ; // ❌ MultipleVariableDeclarations + spacing
            while( ldividend >= ( temp << 1 ) ){ // ❌ ParenPad
                temp = temp << 1; multiple = multiple << 1;  // ❌ OneStatementPerLine
            }
            ldividend -= temp; result += multiple; // ❌ OneStatementPerLine
        }
        if( negative ){ result = -result; } // ❌ Unnecessary parentheses + OneStatementPerLine
        if( result > Integer.MAX_VALUE ){ return Integer.MAX_VALUE; } else if( result < Integer.MIN_VALUE ){ return Integer.MIN_VALUE; } // ❌ multiple statements + line too long
        return (int) result ; // ❌ extra space before semicolon
    }

    public static void main( String[] args ){ // ❌ ParenPad
        System.out.println( divide( 10 , 3 ) ); // ❌ extra spaces + ParenPad
    }
}
